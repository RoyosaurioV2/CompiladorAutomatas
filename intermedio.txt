Inicio:
    # Valores fijos para demostrar optimización
    = 15 @num1
    = 10 @num2
    = 5 @num3

    # Operaciones constantes
    + 3 5 t1          # Optimizable: 3+5=8
    * t1 2 t2         # Optimizable: 8*2=16

    # Comparaciones redundantes
    > @num1 @num2 t3  # 15 > 10 → true
    > @num1 @num2 t4  # Redundante - misma comparación

    # Código muerto potencial
    = 100 t5          # Temporal no usado
    = "Resultado: " t6
    Esc t6

    # Expresión booleana constante
    && true false t7   # Optimizable: false

    # Bloque condicional con constante
    IrSino t7 Fin      # Salto nunca tomado

    = "Este mensaje no aparece" t8
    Esc t8

Fin:
    = "Fin del programa" t9
    Esc t9
